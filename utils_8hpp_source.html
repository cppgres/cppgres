37template <typename T> constexpr std::string_view type_name() {
38#ifdef __clang__
39 constexpr std::string_view p = __PRETTY_FUNCTION__;
40 constexpr std::string_view key = "T = ";
41 const auto start = p.find(key) + key.size();
42 const auto end = p.find(']', start);
43 return p.substr(start, end - start);
44#elif defined(__GNUC__)
45 constexpr std::string_view p = __PRETTY_FUNCTION__;
46 constexpr std::string_view key = "T = ";
47 const auto start = p.find(key) + key.size();
48 const auto end = p.find(';', start);
49 return p.substr(start, end - start);
50#elif defined(_MSC_VER)
51 constexpr std::string_view p = __FUNCSIG__;
52 constexpr std::string_view key = "type_name<";
53 const auto start = p.find(key) + key.size();
54 const auto end = p.find(">(void)");
55 return p.substr(start, end - start);
56#else
57 return "Unsupported compiler";
58#endif
59}
