198 static datum into_datum(const std::string_view &t) {
199 size_t sz = t.size();
200 void *result = ffi_guard{::palloc}(sz + VARHDRSZ);
201 SET_VARSIZE(result, t.size() + VARHDRSZ);
202 memcpy(VARDATA(result), t.data(), sz);
203 return datum(reinterpret_cast<::Datum>(result));
204 }
