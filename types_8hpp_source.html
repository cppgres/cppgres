166 static datum into_datum(const std::string_view &t) {
167 size_t sz = t.size();
168 void *result = ffi_guarded(::palloc)(sz + VARHDRSZ);
169 SET_VARSIZE(result, t.size() + VARHDRSZ);
170 memcpy(VARDATA(result), t.data(), sz);
171 return datum(reinterpret_cast<::Datum>(result));
172 }
